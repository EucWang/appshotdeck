# Light Pad (Adjust Light) Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Replace the "Adjust Light" slider with a 2D light pad (draggable pad) that controls shadow direction and intensity, matching shots.so's approach. The light pad lets users drag a point on a 2D grid to set `shadowPercentX` / `shadowPercentY`, which dynamically controls box-shadow offset, blur, and spread on flat frames, and light position on 3D frames.

**Architecture:** Replace `frameLightIntensity: number` (0–200) with `shadowPercentX: number` (-50 to 50) and `shadowPercentY: number` (-50 to 50) on the `Slide` type. The StylePanel's "Adjust Light" section becomes a popup containing a 2D draggable pad + shadow opacity slider. `computeShadow()` in `mockupStyle.ts` is rewritten to accept percentX/percentY and produce multi-layer directional box-shadows. `Device3D.tsx` moves the directional light position based on percentX/percentY.

**Tech Stack:** React 19, TypeScript, Zustand (persist), Tailwind CSS v3, Three.js (3D light), lucide-react icons, react-i18next.

```typescript
export function computeShadow(
  mode: ShadowMode,
  percentX: number = 0,
  percentY: number = -20
): string | null {
  if (mode === 'none') return null
  if (mode === 'adaptive') return null

  const distance = Math.sqrt(percentX * percentX + percentY * percentY)
  const maxBlur = mode === 'spread' ? 80 : 24
  const maxOffset = mode === 'spread' ? 48 : 8

  const blur = Math.max(maxBlur * distance / 50, 3)
  const factor = distance / 50
  const offsetX = percentX * factor * (mode === 'spread' ? 1.2 : 0.4)
  const offsetY = percentY * factor * (mode === 'spread' ? 1.2 : 0.4)
  const alpha = 0.25 + factor * 0.35

  const layers = [
    { x: offsetX * 0.15, y: offsetY * 0.15, blur: blur * 0.5, alpha: alpha * 0.9 },
    { x: offsetX * 0.4,  y: offsetY * 0.4,  blur: blur * 0.8, alpha: alpha * 0.6 },
    { x: offsetX * 0.7,  y: offsetY * 0.7,  blur: blur * 1.0, alpha: alpha * 0.4 },
    { x: offsetX * 1.0,  y: offsetY * 1.0,  blur: blur * 1.5, alpha: alpha * 0.25 },
  ]

  return layers
    .map(l => `${Math.round(l.x)}px ${Math.round(l.y)}px ${Math.round(l.blur)}px rgba(0,0,0,${l.alpha.toFixed(2)})`)
    .join(', ')
}
```

### Task 5: Create `LightPad` component

**Files:**
- Create: `src/components/Sidebar/LightPad.tsx`

This is a 2D draggable pad (like shots.so's `rz` component). The user drags a small circle on a square grid. The position maps to `shadowPercentX` (-50 to 50) and `shadowPercentY` (-50 to 50). Y axis is inverted (dragging up = negative Y = shadow goes down).

```tsx
import { useCallback, useRef, useState } from 'react'

interface Props {
  valueX: number
  valueY: number
  onChange: (x: number, y: number) => void
  size?: number
}

export function LightPad({ valueX, valueY, onChange, size = 140 }: Props) {
  const padRef = useRef<HTMLDivElement>(null)
  const [dragging, setDragging] = useState(false)

  const updateFromEvent = useCallback((clientX: number, clientY: number) => {
    const el = padRef.current
    if (!el) return
    const rect = el.getBoundingClientRect()
    const padSize = rect.width
    const cx = clientX - rect.left
    const cy = clientY - rect.top
    const x = Math.max(-50, Math.min(50, ((cx / padSize) - 0.5) * 100))
    const y = Math.max(-50, Math.min(50, ((cy / padSize) - 0.5) * 100))
    onChange(Math.round(x), Math.round(y))
  }, [onChange])

  const handlePointerDown = useCallback((e: React.PointerEvent) => {
    e.preventDefault()
    e.stopPropagation()
    setDragging(true)
    updateFromEvent(e.clientX, e.clientY)
    const el = e.currentTarget as HTMLElement
    el.setPointerCapture(e.pointerId)
  }, [updateFromEvent])

  const handlePointerMove = useCallback((e: React.PointerEvent) => {
    if (!dragging) return
    e.preventDefault()
    updateFromEvent(e.clientX, e.clientY)
  }, [dragging, updateFromEvent])

  const handlePointerUp = useCallback(() => {
    setDragging(false)
  }, [])

  const handleReset = useCallback((e: React.MouseEvent) => {
    e.stopPropagation()
    onChange(0, -20)
  }, [onChange])

  const knobX = ((valueX / 100) + 0.5) * size
  const knobY = ((valueY / 100) + 0.5) * size
  const pad = 10
  const inner = size - pad * 2

  return (
    <div className="flex flex-col items-center gap-2">
      <div
        ref={padRef}
        className="relative rounded-xl bg-black/5 dark:bg-white/10 border border-black/10 dark:border-white/15 cursor-crosshair touch-none"
        style={{ width: size, height: size }}
        onPointerDown={handlePointerDown}
        onPointerMove={handlePointerMove}
        onPointerUp={handlePointerUp}
      >
        {/* Grid lines */}
        <svg
          width={size}
          height={size}
          style={{ position: 'absolute', inset: 0, pointerEvents: 'none' }}
        >
          {[25, 50, 75].map((pct) => (
            <g key={pct}>
              <line x1={size * pct / 100} y1={0} x2={size * pct / 100} y2={size} stroke="currentColor" strokeWidth={0.5} className="text-black/10 dark:text-white/10" />
              <line x1={0} y1={size * pct / 100} x2={size} y2={size * pct / 100} stroke="currentColor" strokeWidth={0.5} className="text-black/10 dark:text-white/10" />
            </g>
          ))}
          {/* Center crosshair */}
          <line x1={size / 2} y1={0} x2={size / 2} y2={size} stroke="currentColor" strokeWidth={1} strokeDasharray="3 3" className="text-black/20 dark:text-white/20" />
          <line x1={0} y1={size / 2} x2={size} y2={size / 2} stroke="currentColor" strokeWidth={1} strokeDasharray="3 3" className="text-black/20 dark:text-white/20" />
        </svg>
        {/* Knob */}
        <div
          className="absolute rounded-full bg-indigo-500 border-2 border-white shadow-md"
          style={{
            width: 14,
            height: 14,
            left: knobX - 7,
            top: knobY - 7,
            transition: dragging ? 'none' : 'left 0.1s, top 0.1s',
          }}
        />
      </div>
      <button
        onClick={handleReset}
        className="text-xs text-muted hover:text-foreground transition-colors"
      >
        Reset
      </button>
    </div>
  )
}
```
